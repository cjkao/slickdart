to use cell span, there will need a MetaList class and meta function to provide span condition 


 var metaList = new grid.MetaList(_data, getMeta);
 /// specify column id or field name
 Map getMeta(int row) {
   if (row == 1 )
     return {
     grid.MetaList.COLUMN: {'duration': 2}  //'duration' is field name
   };
   return {};
 }
